17/// Cuboid primitive
18///
19/// This is a cuboid primitive that can be used in various applications.
20struct Cuboid {
21    half_size: vec3<f32>,
22}
23
24fn create(x_length: f32, y_length: f32, z_length: f32) -> Cuboid {
25    return from_size(vec3<f32>(x_length, y_length, z_length));
26}
