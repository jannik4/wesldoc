math_utils 
1import package::constants::PI;
2import primitives::{cuboid::Cuboid, cylinder::Cylinder};
3
4/// Calculates the volume of a [`Cuboid`].
5fn cuboid_volume(cuboid: Cuboid) -> f32 {
6    return cuboid.half_size.x * cuboid.half_size.y * cuboid.half_size.z * 8.0;
7}
8
9/// Calculates the volume of a [`Cylinder`].
10fn cylinder_volume(cylinder: Cylinder) -> f32 {
11    let base_area = PI * cylinder.radius * cylinder.radius;
12    return base_area * cylinder.half_height * 2.0;
13}
